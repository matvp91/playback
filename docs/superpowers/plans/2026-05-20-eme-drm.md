# EME / DRM Support Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add EME support for Widevine, PlayReady, and FairPlay on MSE, with manifest-driven key system selection and app-delegated license acquisition through the existing `NetworkService`.

**Architecture:** Extend the DASH parser to read `<ContentProtection>` into a new `Protection` model on `SwitchingSet`. Extend the existing `mediaCapabilities.decodingInfo` probe with `keySystemConfiguration` to select a supported key system per stream. A new `EmeController` (sibling to `BufferController`, `StreamController`) owns the `MediaKeys` lifecycle and dispatches license requests via `NetworkService` using a new `NetworkRequestType.LICENSE`. FairPlay defers `setMediaKeys` until the `encrypted` event; Widevine/PlayReady drive sessions from manifest PSSH. No gating between controllers — the EME spec allows `setMediaKeys` in any order relative to MSE attach.

**Tech Stack:** TypeScript, Vitest (happy-dom), pnpm workspace. Tests in `packages/cmaf-lite/test/`. Spec: [docs/superpowers/specs/2026-05-20-eme-drm-design.md](../specs/2026-05-20-eme-drm-design.md).

- [ ] **Step 1: Create `protection.ts` with the public entry point**

```ts
// packages/cmaf-lite/lib/dash/protection.ts
import type * as txml from "txml";
import type { KeySystemInfo, Protection } from "../types/manifest";
import { KeySystem } from "../types/media";
import * as XmlUtils from "../utils/xml_utils";

const MP4_PROTECTION_SCHEME = "urn:mpeg:dash:mp4protection:2011";

const KEY_SYSTEM_BY_UUID: Record<string, KeySystem> = {
  "edef8ba9-79d6-4ace-a3c8-27dcd51d21ed": KeySystem.WIDEVINE,
  "9a04f079-9840-4286-ab92-e65be0885f95": KeySystem.PLAYREADY,
  "94ce86fb-07ff-4f43-adb8-93d2fa968ca2": KeySystem.FAIRPLAY,
};

/**
 * Reads all `<ContentProtection>` elements on an AdaptationSet
 * and returns a `Protection` model, or `null` when no protection
 * is signalled.
 */
export function readProtection(adaptationSet: txml.TNode): Protection | null {
  const elements = XmlUtils.children(adaptationSet, "ContentProtection");
  if (elements.length === 0) return null;

  let scheme: "cenc" | "cbcs" | null = null;
  let defaultKid: string | null = null;
  const keySystems: Partial<Record<KeySystem, KeySystemInfo>> = {};

  for (const el of elements) {
    const schemeIdUri = XmlUtils.attr(el, "schemeIdUri", XmlUtils.parseString);
    if (!schemeIdUri) continue;

    if (schemeIdUri === MP4_PROTECTION_SCHEME) {
      const value = XmlUtils.attr(el, "value", XmlUtils.parseString);
      if (value === "cenc" || value === "cbcs") scheme = value;
      const kid = XmlUtils.attr(el, "cenc:default_KID", XmlUtils.parseString);
      if (kid) defaultKid = kid.toLowerCase();
      continue;
    }

    const uuid = extractUuid(schemeIdUri);
    if (!uuid) continue;
    const keySystem = KEY_SYSTEM_BY_UUID[uuid];
    if (!keySystem) continue;

    keySystems[keySystem] = readKeySystemInfo(el, keySystem);
  }

  if (scheme === null) return null;
  if (defaultKid === null) {
    throw new Error("ContentProtection: mp4protection present without cenc:default_KID");
  }
  return { scheme, defaultKid, keySystems };
}

function extractUuid(schemeIdUri: string): string | null {
  const match = /^urn:uuid:([0-9a-f-]+)$/i.exec(schemeIdUri);
  return match ? match[1]!.toLowerCase() : null;
}

function readKeySystemInfo(el: txml.TNode, keySystem: KeySystem): KeySystemInfo {
  if (keySystem === KeySystem.FAIRPLAY) {
    // FairPlay carries the content id in the `value` attribute on
    // some packagers and in a child element on others. Support both.
    const valueAttr = XmlUtils.attr(el, "value", XmlUtils.parseString);
    if (valueAttr?.startsWith("skd://")) return { contentId: valueAttr };
    const child = XmlUtils.children(el, "cenc:pssh")[0];
    const text = child ? XmlUtils.text(child) : undefined;
    if (text?.startsWith("skd://")) return { contentId: text };
    return {};
  }

  const psshNode = XmlUtils.children(el, "cenc:pssh")[0];
  if (!psshNode) return {};
  const base64 = XmlUtils.text(psshNode);
  if (!base64) return {};
  return { pssh: base64ToBytes(base64.trim()) };
}

function base64ToBytes(b64: string): Uint8Array {
  const bin = atob(b64);
  const out = new Uint8Array(bin.length);
  for (let i = 0; i < bin.length; i++) out[i] = bin.charCodeAt(i);
  return out;
}
```

import { readProtection } from "../../lib/dash/protection";

    expect(p.scheme).toBe("cenc");
    expect(p.keySystems).toEqual({});
  });
});

```ts
import { PROP_DECODING_INFO, PROP_HIERARCHY, PROP_KEY_SYSTEM_ACCESS } from "../constants";
import type { DrmConfig } from "../config";
import { KeySystem } from "../types/media";

type DecodingProbe = {
  info: MediaCapabilitiesDecodingInfo;
  keySystemAccess?: MediaKeySystemAccess;
};

async function probeDecodingInfo(
  codec: string,
  track: Track,
  switchingSet: SwitchingSet,
  drm: DrmConfig,
): Promise<DecodingProbe> {
  const candidates = candidateKeySystems(switchingSet, drm);
  if (candidates.length === 0) {
    const info = await probeOnce(codec, track, undefined);
    return { info };
  }
  for (const keySystem of candidates) {
    const config = buildKeySystemConfig(keySystem, switchingSet, track);
    const info = await probeOnce(codec, track, config);
    if (info.supported) {
      return { info, keySystemAccess: info.keySystemAccess ?? undefined };
    }
  }
  return { info: { supported: false, smooth: false, powerEfficient: false, keySystemAccess: null } };
}

function candidateKeySystems(switchingSet: SwitchingSet, drm: DrmConfig): KeySystem[] {
  if (!switchingSet.protection) return [];
  return drm.preferredKeySystems.filter((ks) => ks in switchingSet.protection!.keySystems);
}

function buildKeySystemConfig(
  keySystem: KeySystem,
  switchingSet: SwitchingSet,
  track: Track,
): MediaKeySystemConfiguration {
  const contentType =
    track.type === MediaType.VIDEO
      ? `video/mp4; codecs="${switchingSet.codec}"`
      : `audio/mp4; codecs="${switchingSet.codec}"`;
  const cap = { contentType, robustness: defaultRobustness(keySystem) };
  return {
    initDataTypes: ["cenc"],
    distinctiveIdentifier: "optional",
    persistentState: "optional",
    sessionTypes: ["temporary"],
    [track.type === MediaType.VIDEO ? "videoCapabilities" : "audioCapabilities"]: [cap],
  };
}

function defaultRobustness(keySystem: KeySystem): string {
  if (keySystem === KeySystem.WIDEVINE) return "SW_SECURE_CRYPTO";
  if (keySystem === KeySystem.PLAYREADY) return "150";
  return ""; // FairPlay: leave empty
}

async function probeOnce(
  codec: string,
  track: Track,
  keySystemConfiguration: MediaKeySystemConfiguration | undefined,
): Promise<MediaCapabilitiesDecodingInfo> {
  const base: MediaDecodingConfiguration =
    track.type === MediaType.VIDEO
      ? {
          type: "media-source",
          video: {
            contentType: `video/mp4; codecs="${codec}"`,
            width: track.width,
            height: track.height,
            bitrate: track.bandwidth,
            framerate: 30,
          },
        }
      : {
          type: "media-source",
          audio: {
            contentType: `audio/mp4; codecs="${codec}"`,
            bitrate: track.bandwidth,
            channels: "2",
            samplerate: 48000,
          },
        };
  if (keySystemConfiguration) base.keySystemConfiguration = keySystemConfiguration as any;
  return navigator.mediaCapabilities.decodingInfo(base);
}

// packages/cmaf-lite/lib/utils/playready_utils.ts

/**
 * Unwraps a PlayReady CDM license challenge.
 *
 * PlayReady challenges arrive as UTF-16-LE encoded XML wrapping a
 * base64-encoded inner SOAP body. License servers expect the raw
 * inner body, not the envelope.
 *
 * Returns the original buffer unchanged when it does not match the
 * PlayReady envelope shape (some content/CDM combinations emit the
 * SOAP body directly).
 */
export function unwrapPlayReadyChallenge(buffer: ArrayBuffer): ArrayBuffer {
  if (buffer.byteLength < 2) return buffer;
  const xml = new TextDecoder("utf-16le").decode(buffer);
  const match = /<Challenge[^>]*>([^<]+)<\/Challenge>/.exec(xml);
  if (!match) return buffer;
  const bin = atob(match[1]!);
  const out = new Uint8Array(bin.length);
  for (let i = 0; i < bin.length; i++) out[i] = bin.charCodeAt(i);
  return out.buffer;
}

// packages/cmaf-lite/lib/media/eme_controller.ts
import type { Player } from "../player";
import { Events } from "../events";
import { PROP_KEY_SYSTEM_ACCESS } from "../constants";
import type { Manifest } from "../types/manifest";
import { KeySystem, MediaType } from "../types/media";
import { NetworkRequestType } from "../types/net";
import { unwrapPlayReadyChallenge } from "../utils/playready_utils";

const TEXT_DECODER = new TextDecoder();

/**
 * Owns the MediaKeys lifecycle for protected presentations. Dormant
 * for clear content.
 */
export class EmeController {
  private manifest_: Manifest | null = null;
  private media_: HTMLMediaElement | null = null;
  private mediaKeys_: MediaKeys | null = null;
  private mediaKeysAttached_ = false;
  private keySystem_: KeySystem | null = null;
  private activeSessions_ = new Set<MediaKeySession>();
  private psshSeen_ = new Set<string>();
  private onEncrypted_: ((ev: MediaEncryptedEvent) => void) | null = null;

  constructor(private player_: Player) {
    player_.on(Events.MANIFEST_LOADING, this.handleManifestLoading_);
    player_.on(Events.STREAMS_CREATED, this.handleStreamsCreated_);
    player_.on(Events.MEDIA_ATTACHED, this.handleMediaAttached_);
    player_.on(Events.MEDIA_DETACHING, this.handleMediaDetaching_);
  }

  destroy() {
    this.teardown_().catch(() => {});
    this.player_.off(Events.MANIFEST_LOADING, this.handleManifestLoading_);
    this.player_.off(Events.STREAMS_CREATED, this.handleStreamsCreated_);
    this.player_.off(Events.MEDIA_ATTACHED, this.handleMediaAttached_);
    this.player_.off(Events.MEDIA_DETACHING, this.handleMediaDetaching_);
  }

  private handleManifestLoading_ = () => {
    this.manifest_ = null;
  };

  private handleStreamsCreated_ = () => {
    this.manifest_ = this.player_.getManifest();
    this.maybeActivate_();
  };

  private handleMediaAttached_ = ({ media }: { media: HTMLMediaElement }) => {
    this.media_ = media;
    this.maybeActivate_();
  };

  private handleMediaDetaching_ = async () => {
    await this.teardown_();
  };

  private maybeActivate_() {
    if (!this.manifest_ || !this.media_) return;
    const access = this.findKeySystemAccess_();
    if (!access) return;
    void this.activate_(access);
  }

  private findKeySystemAccess_(): MediaKeySystemAccess | null {
    if (!this.manifest_) return null;
    for (const type of [MediaType.VIDEO, MediaType.AUDIO]) {
      const streams = this.player_.getStreams(type);
      for (const stream of streams) {
        const access = (stream as any)[PROP_KEY_SYSTEM_ACCESS] as
          | MediaKeySystemAccess
          | undefined;
        if (access) return access;
      }
    }
    return null;
  }

  private async activate_(access: MediaKeySystemAccess) {
    if (this.mediaKeys_) return; // already activated
    try {
      this.keySystem_ = access.keySystem as KeySystem;
      this.mediaKeys_ = await access.createMediaKeys();

      const cert = this.player_.getConfig().drm.serverCertificates[this.keySystem_];
      if (cert) await this.mediaKeys_.setServerCertificate(cert);

      if (this.keySystem_ === KeySystem.FAIRPLAY) {
        this.attachEncryptedListener_();
      } else {
        await this.attachMediaKeys_();
        this.createSessionsFromManifest_();
      }
    } catch (err) {
      this.emitError_(err);
    }
  }

  private async attachMediaKeys_() {
    if (this.mediaKeysAttached_ || !this.media_ || !this.mediaKeys_) return;
    await this.media_.setMediaKeys(this.mediaKeys_);
    this.mediaKeysAttached_ = true;
  }

  private attachEncryptedListener_() {
    if (!this.media_) return;
    this.onEncrypted_ = (event) => {
      void this.handleEncryptedEvent_(event);
    };
    this.media_.addEventListener("encrypted", this.onEncrypted_);
  }

  private async handleEncryptedEvent_(event: MediaEncryptedEvent) {
    try {
      await this.attachMediaKeys_();
      if (!event.initData) return;
      await this.createSession_(event.initDataType, new Uint8Array(event.initData));
    } catch (err) {
      this.emitError_(err);
    }
  }

  private createSessionsFromManifest_() {
    if (!this.manifest_ || !this.keySystem_) return;
    for (const ss of this.manifest_.switchingSets) {
      const info = ss.protection?.keySystems[this.keySystem_];
      if (!info?.pssh) continue;
      const fingerprint = bytesFingerprint(info.pssh);
      if (this.psshSeen_.has(fingerprint)) continue;
      this.psshSeen_.add(fingerprint);
      void this.createSession_("cenc", info.pssh);
    }
  }

  private async createSession_(initDataType: string, initData: Uint8Array) {
    if (!this.mediaKeys_ || !this.keySystem_) return;
    const session = this.mediaKeys_.createSession("temporary");
    this.activeSessions_.add(session);

    session.addEventListener("message", (ev) => {
      void this.handleSessionMessage_(session, ev as MediaKeyMessageEvent);
    });
    session.addEventListener("keystatuseschange", () => {
      this.handleKeyStatusesChange_(session);
    });

    await session.generateRequest(initDataType, initData);

    this.player_.emit(Events.KEY_SESSION_CREATED, {
      keySystem: this.keySystem_,
      sessionId: session.sessionId,
    });
  }

  private async handleSessionMessage_(
    session: MediaKeySession,
    event: MediaKeyMessageEvent,
  ) {
    if (!this.keySystem_) return;
    try {
      let body: BodyInit = event.message;
      if (this.keySystem_ === KeySystem.PLAYREADY) {
        body = unwrapPlayReadyChallenge(event.message);
      }
      const url = this.player_.getConfig().drm.licenseUrls[this.keySystem_];
      if (!url) throw new Error(`No license URL configured for ${this.keySystem_}`);

      const request = this.player_.getNetworkService().request(
        NetworkRequestType.LICENSE,
        url,
        undefined,
        { method: "POST", body },
      );
      const response = await request.promise;
      if (response === undefined || response === null || !("data" in response)) {
        throw new Error("License request aborted");
      }
      await session.update(response.data);
    } catch (err) {
      this.emitError_(err);
    }
  }

  private handleKeyStatusesChange_(session: MediaKeySession) {
    const statuses = new Map<string, MediaKeyStatus>();
    session.keyStatuses.forEach((status, keyId) => {
      statuses.set(bytesFingerprint(new Uint8Array(keyId as ArrayBuffer)), status);
    });
    this.player_.emit(Events.KEY_STATUSES_CHANGED, {
      sessionId: session.sessionId,
      statuses,
    });
    for (const status of statuses.values()) {
      if (status === "internal-error" || status === "output-restricted") {
        this.emitError_(new Error(`Fatal key status: ${status}`));
        return;
      }
    }
  }

  private emitError_(err: unknown) {
    // Existing player error surface is `console.error`-style today; once
    // a typed ERROR event lands this becomes a single emit. For now log.
    console.error("[EmeController]", err);
  }

  private async teardown_() {
    if (this.onEncrypted_ && this.media_) {
      this.media_.removeEventListener("encrypted", this.onEncrypted_);
      this.onEncrypted_ = null;
    }
    for (const session of this.activeSessions_) {
      try {
        await session.close();
      } catch {
        /* ignore */
      }
    }
    this.activeSessions_.clear();
    this.psshSeen_.clear();
    if (this.media_ && this.mediaKeysAttached_) {
      try {
        await this.media_.setMediaKeys(null);
      } catch {
        /* ignore */
      }
    }
    this.mediaKeys_ = null;
    this.mediaKeysAttached_ = false;
    this.keySystem_ = null;
    this.manifest_ = null;
    this.media_ = null;
  }
}

function bytesFingerprint(bytes: Uint8Array): string {
  let hex = "";
  for (let i = 0; i < bytes.length; i++) {
    hex += bytes[i]!.toString(16).padStart(2, "0");
  }
  return hex;
}
```
